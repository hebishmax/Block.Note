القالب 1: محفظة مركزية BSC

هذا العقد الذكي يقوم بجمع أي تبرعات تُرسل إليه، ثم يحولها مباشرةً إلى محفظة مركزية تعمل على شبكة BSC.

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract DonationCollector {
    // تحديد المحفظة المركزية باستخدام العنوان المعطى
    address payable public centralWallet = payable(0x61916c290c837cd1fe283dcf56801bc3a16f7b6f);

    receive() external payable {
        centralWallet.transfer(msg.value);
    }
}

القالب 2: NFT تعمل على BSC

عقد يقوم بإنشاء NFT واحدة تلقائيًا عند النشر وإرسالها إلى محفظة الأدمن. يعتمد على OpenZeppelin ويُستخدم لتكريم المتبرعين أو إنشاء رموز NFT فريدة.

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;
import "@openzeppelin/contracts/token/ERC721/extensions/ERC721URIStorage.sol";
import "@openzeppelin/contracts/access/Ownable.sol";
contract DonationNFT is ERC721URIStorage, Ownable {
    uint256 public tokenCounter;
    string public defaultTokenURI;
    address public adminWallet;
    constructor(address _adminWallet, string memory _defaultTokenURI) ERC721("DonationNFT", "DNFT") {
        adminWallet = _adminWallet;
        defaultTokenURI = _defaultTokenURI;
        tokenCounter = 0;
        _safeMint(adminWallet, tokenCounter);
        _setTokenURI(tokenCounter, defaultTokenURI);
        tokenCounter++;
    }
}
